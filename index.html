function render(data) { const container = document.getElementById('directory'); container.innerHTML = ''; Object.keys(data).sort().forEach(loc => { const section = document.createElement('div'); section.className = 'location-section'; const header = document.createElement('div'); header.className = 'location-header'; header.innerHTML = `${loc}`; header.onclick = () => toggle(header); const content = document.createElement('div'); content.className = 'location-content'; data[loc].sort((a,b) => a.dept.localeCompare(b.dept)).forEach(c => { const dial = c.phone.replace(/\D/g, ''); // We use both href and onclick to ensure iOS recognizes human intent const callAction = `window.location.href='tel:${dial}'; return false;`; content.innerHTML += `
${c.dept}

${c.name}

${c.title}

Call - ${c.phone}
📞
`; }); section.appendChild(header); section.appendChild(content); container.appendChild(section); }); }
